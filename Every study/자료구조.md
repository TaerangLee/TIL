## 삭제

    void doDelete(int deleteData) {

	    if (numOfDatas == 0) {
	    	printf("리스트에 데이터가 없음");
	    	return;
	    }

        for (int i = 0; i < numOfDatas; i++) {
            if (list[i] == deleteData) {
                for (int j = i; j < numOfDatas - 1; j++) {
                    list[j] = list[j + 1]; 
                }
                numOfDatas -= 1;
                i -= 1;
            }

        }
    }
